Used alongside [useDashboardReport](/react/use-dashboard-report) for fully customizable dashboards. Below is an example with shadcn showing fully custom styling.

```tsx App.tsx
import {
  QuillProvider,
  useDashboard,
  useDashboardReport,
  StaticChart,
} from "@quillsql/react";
import { Card, CardHeader } from "@/components/ui/card";
import { Skeleton } from "@/components/ui/skeleton";

function App() {
  return (
    <QuillProvider
      tenants={[{ tenantField: "customer_id", tenantIds: [2] }]}
      publicKey="6579031b3e41c378aa8180ec"
    >
      <CustomDashboard />
    </QuillProvider>
  );
}

function CustomDashboard() {
  const { sections, isLoading } = useDashboard("quill demo dashboard");

  return (
    <>
      <ChartsSection reports={sections["charts"]} />
    </>
  );
}

function ChartsSection({ reports }: { reports: any[] }) {
  return (
    <div className="grid grid-cols-1 lg:grid-cols-6 gap-6">
      {reports.map((report: any) => (
        <ChartCard reportId={report.id} name={report.name} />
      ))}
    </div>
  );
}

function ChartCard({ reportId, name }) {
  const { report, loading } = useDashboardReport(reportId);
  if (loading) {
    return (
      <div className="lg:col-span-1">
        <Card
          className="h-full shadow-none bg-transparent border-none"
          title={name}
        >
          <Skeleton />
        </Card>
      </div>
    );
  }
  return (
    <div className="lg:col-span-1">
      <Card
        className="h-full shadow-none bg-transparent border-none"
        title={name}
      >
        <StaticChart reportId={report.id} />
      </Card>
    </div>
  );
}
```

## Working with filters

`useDashboard` returns a `filters` array describing the dashboard's available filters and an `applyFilters` function that replaces the dashboard's filter set on every call. Each entry in `filters` has a `type` of `select`, `multiselect`, `date`, or `tenant`.

Because `applyFilters` replaces the filter set, the typical pattern is to map over the current `filters`, merge in an override for the one filter that changed, and pass the full result to `applyFilters`. This is the same pattern used by the built-in `Dashboard` component in `@quillsql/react`.

### Select, multiselect, and date filters

```tsx App.tsx
import {
  QuillProvider,
  useDashboard,
  StaticChart,
  DashboardFilterType,
} from "@quillsql/react";

function App() {
  return (
    <QuillProvider
      tenants={[{ tenantField: "customer_id", tenantIds: [2] }]}
      publicKey="6579031b3e41c378aa8180ec"
    >
      <FilteredDashboard name="quill demo dashboard" />
    </QuillProvider>
  );
}

function FilteredDashboard({ name }: { name: string }) {
  const { sections, filters, applyFilters, isLoading } = useDashboard(name);

  // Map current filters into the applyFilters payload, overriding the one
  // that changed. Mirrors the @quillsql/react Dashboard component.
  const applyDashboardFilters = (
    overrides: Record<
      string,
      string | string[] | { startDate?: Date; endDate?: Date; preset?: string }
    >,
  ) => {
    const payload = filters.map((filter) => {
      if (filter.type === DashboardFilterType.Date) {
        const override = overrides[filter.label] as
          | { startDate?: Date; endDate?: Date; preset?: string }
          | undefined;
        return {
          label: filter.label,
          value: override ?? {
            startDate: filter.value.startDate,
            endDate: filter.value.endDate,
            preset: filter.value.presetValue ?? "",
          },
        };
      }
      const override = overrides[filter.label] as string | string[] | undefined;
      return { label: filter.label, value: override ?? filter.value };
    });
    applyFilters(payload);
  };

  return (
    <div>
      <div style={{ display: "flex", gap: 12, marginBottom: 16 }}>
        {filters.map((filter) => {
          // Select filter — value is a string
          if (filter.type === DashboardFilterType.Select) {
            return (
              <select
                key={filter.label}
                value={filter.value ?? ""}
                onChange={(e) =>
                  applyDashboardFilters({ [filter.label]: e.target.value })
                }
                disabled={isLoading}
              >
                <option value="">All {filter.label}</option>
                {filter.options.map((option) => (
                  <option key={option.value} value={option.value}>
                    {option.label}
                  </option>
                ))}
              </select>
            );
          }

          // Multiselect filter — value is a string[]
          if (filter.type === DashboardFilterType.MultiSelect) {
            return (
              <select
                key={filter.label}
                multiple
                value={Array.isArray(filter.value) ? filter.value : []}
                onChange={(e) => {
                  const next = Array.from(e.target.selectedOptions).map(
                    (o) => o.value,
                  );
                  applyDashboardFilters({ [filter.label]: next });
                }}
                disabled={isLoading}
              >
                {filter.options.map((option) => (
                  <option key={option.value} value={option.value}>
                    {option.label}
                  </option>
                ))}
              </select>
            );
          }

          // Date filter — value is { startDate, endDate, presetValue }
          if (filter.type === DashboardFilterType.Date) {
            return (
              <div key={filter.label} style={{ display: "flex", gap: 8 }}>
                <select
                  value={filter.value.presetValue ?? ""}
                  onChange={(e) => {
                    const option = filter.options.find(
                      (o) => o.value === e.target.value,
                    );
                    applyDashboardFilters({
                      [filter.label]: {
                        startDate: option?.startDate,
                        endDate: option?.endDate,
                        preset: e.target.value,
                      },
                    });
                  }}
                  disabled={isLoading}
                >
                  {filter.options.map((option) => (
                    <option key={option.value} value={option.value}>
                      {option.label}
                    </option>
                  ))}
                </select>
                <input
                  type="date"
                  value={
                    filter.value.startDate?.toISOString().slice(0, 10) ?? ""
                  }
                  onChange={(e) =>
                    applyDashboardFilters({
                      [filter.label]: {
                        startDate: new Date(e.target.value),
                        endDate: filter.value.endDate,
                        preset: "",
                      },
                    })
                  }
                  disabled={isLoading}
                />
                <input
                  type="date"
                  value={filter.value.endDate?.toISOString().slice(0, 10) ?? ""}
                  onChange={(e) =>
                    applyDashboardFilters({
                      [filter.label]: {
                        startDate: filter.value.startDate,
                        endDate: new Date(e.target.value),
                        preset: "",
                      },
                    })
                  }
                  disabled={isLoading}
                />
              </div>
            );
          }

          return null;
        })}
      </div>

      {sections &&
        Object.entries(sections).map(([sectionName, reports]) => (
          <section key={sectionName}>
            {reports.map((report) => (
              <StaticChart key={report.id} reportId={report.id} />
            ))}
          </section>
        ))}
    </div>
  );
}
```

### Filter shapes

| `type`        | `value`                                                              | `options`                                                              |
| ------------- | -------------------------------------------------------------------- | ---------------------------------------------------------------------- |
| `select`      | `string`                                                             | `{ label: string; value: string }[]`                                   |
| `multiselect` | `string[]`                                                           | `{ label: string; value: string }[]`                                   |
| `date`        | `{ presetValue?: string; startDate?: Date; endDate?: Date }`         | `{ label: string; value: string; startDate?: Date; endDate?: Date }[]` |
| `tenant`      | `string \| number \| (string \| number)[] \| null`                   | `{ label: string; value: string \| number }[]`                         |

### applyFilters payload

`applyFilters` accepts an array of `{ label, value }` objects. `label` must match a filter's `label` from the `filters` array, and `value` must match the type expected for that filter:

- `select` — `value: string`
- `multiselect` — `value: string[]`
- `date` — `value: { startDate?: Date; endDate?: Date; preset?: string }`

Because `applyFilters` replaces the entire filter set on every call, always include the current values for any filters you want to keep.
