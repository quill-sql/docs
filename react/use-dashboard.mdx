Used alongside [useDashboardReport](/react/use-dashboard-report) for fully customizable dashboards. Below is an example with shadcn showing fully custom styling.

```tsx App.tsx
import {
  QuillProvider,
  useDashboard,
  useDashboardReport,
  StaticChart,
} from "@quillsql/react";
import { Card, CardHeader } from "@/components/ui/card";
import { Skeleton } from "@/components/ui/skeleton";

function App() {
  return (
    <QuillProvider
      tenants={[{ tenantField: "customer_id", tenantIds: [2] }]}
      publicKey="6579031b3e41c378aa8180ec"
    >
      <CustomDashboard />
    </QuillProvider>
  );
}

function CustomDashboard() {
  const { sections, isLoading } = useDashboard("quill demo dashboard");

  return (
    <>
      <ChartsSection reports={sections["charts"]} />
    </>
  );
}

function ChartsSection({ reports }: { reports: any[] }) {
  return (
    <div className="grid grid-cols-1 lg:grid-cols-6 gap-6">
      {reports.map((report: any) => (
        <ChartCard reportId={report.id} name={report.name} />
      ))}
    </div>
  );
}

function ChartCard({ reportId, name }) {
  const { report, loading } = useDashboardReport(reportId);
  if (loading) {
    return (
      <div className="lg:col-span-1">
        <Card
          className="h-full shadow-none bg-transparent border-none"
          title={name}
        >
          <Skeleton />
        </Card>
      </div>
    );
  }
  return (
    <div className="lg:col-span-1">
      <Card
        className="h-full shadow-none bg-transparent border-none"
        title={name}
      >
        <StaticChart reportId={report.id} />
      </Card>
    </div>
  );
}
```

## Working with filters

`useDashboard` returns a `filters` array describing the dashboard's available filters and an `applyFilters` function for updating them. Each filter has a `type` of `select`, `multiselect`, `date`, or `tenant`. To apply a filter, pass `{ label, value }` objects to `applyFilters`, where `label` matches the filter's `label` and `value` matches the type expected for that filter.

### Select, multiselect, and date filters

The example below renders each filter type using its own UI control and updates the dashboard via `applyFilters`. It tracks all current filter values in local state so each control can update independently without clearing the others.

```tsx App.tsx
import { useMemo, useState } from "react";
import {
  QuillProvider,
  useDashboard,
  StaticChart,
  DashboardFilterType,
} from "@quillsql/react";

function App() {
  return (
    <QuillProvider
      tenants={[{ tenantField: "customer_id", tenantIds: [2] }]}
      publicKey="6579031b3e41c378aa8180ec"
    >
      <FilteredDashboard name="quill demo dashboard" />
    </QuillProvider>
  );
}

function FilteredDashboard({ name }: { name: string }) {
  const { sections, filters, applyFilters, isLoading } = useDashboard(name);

  // Build the next applyFilters payload by replacing the entry for the
  // edited filter while preserving every other filter's current value.
  const buildPayload = (
    label: string,
    value: string | string[] | { startDate?: Date; endDate?: Date },
  ) =>
    filters.map((f) =>
      f.label === label
        ? { label: f.label, value }
        : {
            label: f.label,
            value:
              f.type === DashboardFilterType.Date
                ? { startDate: f.value.startDate, endDate: f.value.endDate }
                : f.value,
          },
    );

  const selectFilters = filters.filter(
    (f) => f.type === DashboardFilterType.Select,
  );
  const multiSelectFilters = filters.filter(
    (f) => f.type === DashboardFilterType.MultiSelect,
  );
  const dateFilter = useMemo(
    () => filters.find((f) => f.type === DashboardFilterType.Date),
    [filters],
  );

  return (
    <div>
      <div style={{ display: "flex", gap: 12, marginBottom: 16 }}>
        {/* Select filter: single-value <select> */}
        {selectFilters.map((filter) => (
          <select
            key={filter.label}
            value={filter.value ?? ""}
            onChange={(e) =>
              applyFilters(buildPayload(filter.label, e.target.value))
            }
            disabled={isLoading}
          >
            <option value="">All {filter.label}</option>
            {filter.options.map((option) => (
              <option key={option.value} value={option.value}>
                {option.label}
              </option>
            ))}
          </select>
        ))}

        {/* Multiselect filter: multi-value <select multiple>, value is string[] */}
        {multiSelectFilters.map((filter) => (
          <select
            key={filter.label}
            multiple
            value={filter.value ?? []}
            onChange={(e) => {
              const selected = Array.from(e.target.selectedOptions).map(
                (o) => o.value,
              );
              applyFilters(buildPayload(filter.label, selected));
            }}
            disabled={isLoading}
          >
            {filter.options.map((option) => (
              <option key={option.value} value={option.value}>
                {option.label}
              </option>
            ))}
          </select>
        ))}

        {/* Date filter: presets via options, or raw start/end dates */}
        {dateFilter && (
          <>
            <select
              value={dateFilter.value.presetValue ?? ""}
              onChange={(e) => {
                const preset = dateFilter.options.find(
                  (o) => o.value === e.target.value,
                );
                if (!preset) return;
                applyFilters(
                  buildPayload(dateFilter.label, {
                    startDate: preset.startDate,
                    endDate: preset.endDate,
                  }),
                );
              }}
              disabled={isLoading}
            >
              {dateFilter.options.map((option) => (
                <option key={option.value} value={option.value}>
                  {option.label}
                </option>
              ))}
            </select>

            <input
              type="date"
              value={dateFilter.value.startDate?.toISOString().slice(0, 10) ?? ""}
              onChange={(e) =>
                applyFilters(
                  buildPayload(dateFilter.label, {
                    startDate: new Date(e.target.value),
                    endDate: dateFilter.value.endDate,
                  }),
                )
              }
              disabled={isLoading}
            />
            <input
              type="date"
              value={dateFilter.value.endDate?.toISOString().slice(0, 10) ?? ""}
              onChange={(e) =>
                applyFilters(
                  buildPayload(dateFilter.label, {
                    startDate: dateFilter.value.startDate,
                    endDate: new Date(e.target.value),
                  }),
                )
              }
              disabled={isLoading}
            />
          </>
        )}
      </div>

      {sections &&
        Object.entries(sections).map(([sectionName, reports]) => (
          <section key={sectionName}>
            {reports.map((report) => (
              <StaticChart key={report.id} reportId={report.id} />
            ))}
          </section>
        ))}
    </div>
  );
}
```

### Filter shapes

Each entry in `filters` is one of the following shapes:

| `type`        | `value`                                                              | `options`                                                          |
| ------------- | -------------------------------------------------------------------- | ------------------------------------------------------------------ |
| `select`      | `string`                                                             | `{ label: string; value: string }[]`                               |
| `multiselect` | `string[]`                                                           | `{ label: string; value: string }[]`                               |
| `date`        | `{ presetValue?: string; startDate?: Date; endDate?: Date }`         | `{ label: string; value: string; startDate?: Date; endDate?: Date }[]` |
| `tenant`      | `string \| number \| (string \| number)[] \| null`                   | `{ label: string; value: string \| number }[]`                     |

### applyFilters payload

`applyFilters` accepts an array of `{ label, value }` objects, where `label` matches the filter's `label` from the `filters` array. Use the value type that matches the filter `type`:

- `select` — `value: string`
- `multiselect` — `value: string[]`
- `date` — `value: { startDate?: Date; endDate?: Date; preset?: string }`

`applyFilters` replaces the dashboard's filter set on every call, so always include the current values for the filters you want to keep.
