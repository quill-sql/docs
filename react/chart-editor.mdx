---
title: "ChartEditor"
description: "Edit Quill Reports using a simple form"
---

```tsx App.tsx
import { QuillProvider, ChartEditor } from "@quillsql/react";

function App() {
  return (
    <QuillProvider organizationId="2" publicKey="6579031b3e41c378aa8180ec">
      <ChartEditor
        isOpen={isOpen}
        setIsOpen={setIsOpen}
        reportId="664283fb4db8ad000bfe54d7"
      />
    </QuillProvider>
  );
}
```

Presents the Quill Chart Builder form as a modal.

<Info>
  Make sure `QuillProvider` is a parent of the `ChartEditor` component.
</Info>

## Props

<ParamField path="isOpen" type="boolean" required>
  Whether the ChartEditor is open.
</ParamField>

<ParamField path="setIsOpen" type="boolean" required>
  A callback that sets the open state of the ChartEditor.
</ParamField>

<ParamField path="reportId" type="string" required>
  The id of the report that is being edited.
</ParamField>

<ParamField path="isAdmin" type="boolean">
  Whether the editor has administrative priveleges.
</ParamField>

<ParamField path="chartBuilderTitle" type="string">
  The title of the chart builder form.
</ParamField>

<ParamField path="chartBuilderButtonLabel" type="string">
  The button label for the chart builder.
</ParamField>

<ParamField
  path="onAddToDashboardComplete"
  type="(report: QuillReport) => void"
>
  A callback that is fired when the add to dashboard flow has been completed.
</ParamField>

<ParamField path="onDelete" type="() => void">
  A callback that is fired when the report is deleted.
</ParamField>
