## QuillProvider

<Card>

```tsx App.tsx
import { QuillProvider } from "@quillsql/react";

function App() {
  return (
    <QuillProvider
      publicKey={process.env.QUILL_API_KEY}
      organizationId={user.organizationId}
      theme={THEME}
      queryEndpoint={queryEndpoint}
      queryHeaders={queryHeaders}
    >
      / * the rest of your app * /
    </QuillProvider>
  );
}
```

<ParamField path="publicKey" type="string" required>
  The public Quill API key. This can be found in the Quill portal in the
  "Settings" tab.
</ParamField>

<ParamField path="organizationId" type="string">
  The organization id of the user
</ParamField>

<ParamField path="theme" type="Theme">
  The theme
  <Expandable defaultOpen={false} title="properties">
    <ResponseField name="fontFamily" type="string">
      The font family you want Quill to use. This is usually the same font your app uses. ex: "Inter"
    </ResponseField>

    <ResponseField name="primaryTextColor" type="string">
      The font family you want Quill to use. This is usually the same primary font color your app uses. ex: "#000000"
    </ResponseField>

    <ResponseField name="secondaryTextColor" type="string">
      The font family you want Quill to use. This is usually the same secondary font color your app uses. ex: "#6C727F"
    </ResponseField>

    <ResponseField name="chartLabelColor" type="string">
      The font family you want Quill to use. This is usually the same as or similar to the secondary font your app uses. ex: "#666666"
    </ResponseField>

    <ResponseField name="chartTickColor" type="string">
      The font family you want Quill to use. This is usually the same as or similar to the secondary font your app uses. ex: "#CCCCCC"
    </ResponseField>

    <ResponseField name="chartColors" type="array">
      The chart colors you want Quill to use. This is usually the same as the primary and secondary colors your app uses. You can pass in as many colors as you want, and you can also override these by passing in an array for the colors prop in the Chart component. ex: ['#6269E9',

'#E14F62']

</ResponseField>

  </Expandable>
</ParamField>

<ParamField path="queryEndpoint" type="string">
  The endpoint which uses `quill.query` in your backend server. QuillProvider
  and the components it wraps use this endpiont to access and communicate with
  your database.
</ParamField>

<ParamField path="queryHeaders" type="object">
  The headers needed to allow the requests made by Quill components in your
  application to properly authenticate.
</ParamField>

<ParamField path="withCredentials" type="boolean"></ParamField>

</Card>

## Dashboard

<Card>

```tsx App.tsx
import { QuillProvider, Dashboard } from "@quillsql/react";

function App() {
  return (
    <QuillProvider
      organizationId={organizationId}
      publicKey={API_KEY}
      authToken={user.jwt}
      theme={THEME}
    >
      <Dashboard
        name="spend_dashboard"
        containerStyle={{ width: "100%", height: 400 }}
        onClickDashboardItem={(dashboardItem) => console.log(dashboardItem)}
      />
    </QuillProvider>
  );
}
```

<ParamField path="name" type="string" required>
  The name of the dashboard you created in the Quill Portal. To learn more about
  creating dashboards, learn how to (link) create dashboards in portal.
</ParamField>

<ParamField path="containerStyle" type="React.CSSProperties">
  The CSS styles that wrap the dashboard.
</ParamField>

<ParamField
  path="onClickDashboardItem"
  type="(dashboardItem: DashboardItem) => void"
>
  Callback function that fires when a dashboard item is clicked. A common use case is navigating to a new page based on the dashboardItemId. Dashboard items
  are the list of charts that are in a Dashboard. Note the properties of a
  DashboardItem below:

## useQuill

<Card>

```tsx App.tsx
import { QuillProvider, useQuill } from "@quillsql/react";

function CustomComponent() {
  // pass in any report or chart created in the Quill Portal
  const report = useQuill(QUILL_ID);
  // do anything with report object
  return <div>Custom component</div>;
}

function App() {
  return (
    <QuillProvider
      organizationId={organizationId}
      publicKey={API_KEY}
      authToken={user.jwt}
      theme={THEME}
    >
      <CustomComponent />
    </QuillProvider>
  );
}
```

<ParamField path="name" type="string" required>

## Chart

<Card>

```tsx App.tsx
import { QuillProvider, Chart } from "@quillsql/react";

function App() {
  return (
    <QuillProvider
      organizationId={organizationId}
      publicKey={API_KEY}
      authToken={user.jwt}
      theme={THEME}
    >
      <Chart
        chartId="2348yfge987fq348fg"
        containerStyle={{ width: "100%", height: 400 }}
        colors={["#6269E9", "#E14F62"]}
      />
    </QuillProvider>
  );
}
```

<ParamField path="chartId" type="string" required>
  The chart id. The most usage is through a detail page built to navigate from the dashboard - using onClickDashboardItem to get the dashboardItemId, and navigating to a route (say, reports/:id) where the url param is passed in as the chartId. For a standalone chart, you can find the chartId in the Quill Portal and pass it in directly.

</ParamField>

<ParamField path="containerStyle" type="React.CSSProperties">
  The CSS styles that wrap the chart.
</ParamField>

<ParamField path="colors" type="string[]">
  The chart colors you want Quill to use. Optional because Quill will use your theme colors by default, but this will override the theme colors.

</ParamField>

</Card>
